Get Started

Three steps to your first chart.

1 · Load D3 & billboard.js
<!-- D3 (peer dependency) -->
<script src="https://d3js.org/d3.v6.min.js"></script>

<!-- billboard.js with base style -->
<script src="$YOUR_PATH/billboard.js"></script>
<link rel="stylesheet" href="$YOUR_PATH/billboard.css">
2 · Add a holder & generate
// <div id="chart"></div>

const chart = bb.generate({
  bindto: "#chart",
  render: { mode: "svg" },   // or "canvas"
  data: {
    type: "bar",
    columns: [
      ["data1", 30, 200, 100, 170, 150, 250],
      ["data2", 130, 100, 140, 35, 110, 50]
    ]
  }
});
3 · Or render to canvas high-density
import bb, {scatter, canvas} from "billboard.js/canvas";

bb.generate({
  bindto: "#chart",
  render: { mode: canvas() },   // ← SVG or Canvas
  data: {
    type: scatter(),
    /* thousands of points, still smooth */
  }
});

or install from npm

$ npm install billboard.js        # latest
$ npm install billboard.js@next   # release candidate
